JTunnel is a remote downloader using several features to bypass proxy/fw filtering. Files downloaded are in a "swabbed" format with a unique filename : ooze.swab

To unswab downloaded file you can either :
  • Under Linux : 
    dd if=ooze.swab of="original filename" conv=swab
  • With Java :
    • Use the following pre-compiled jar file : Swab.jar
      • Or
    • Cut and paste the following code on https://www.compilejava.net/
    • Click on "Download Zip"
    • Unzip the downloaed file
    • Use the Swab.jar file : 
              java -jar Swab.jar ooze.swab "original filename"
      
    import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;

public class Swab {
  private static final int BUFFER_SIZE = 4096; // 4KB

  public static void main(String[] args) {
    if (args.length < 2) {
      System.out.println("Please provide input and output files");
      System.exit(0);
    }

    String inputFile = args[0];
    String outputFile = args[1];
    try (
        InputStream inputStream = new FileInputStream(inputFile);
        OutputStream outputStream = new FileOutputStream(outputFile);
    ) {
      byte[] buffer = new byte[BUFFER_SIZE];
      int last_byte;
      int bytesRead;
      while ((bytesRead = inputStream.read(buffer)) != -1) {
        if(bytesRead % 2 == 0) {
          outputStream.write(swab(buffer,bytesRead), 0, bytesRead);
        } else {
          outputStream.write(swab(buffer,bytesRead-1), 0, bytesRead-1);
          last_byte = inputStream.read();
          if(last_byte != -1)
            outputStream.write((byte)last_byte);
          outputStream.write(buffer[bytesRead-1]);
        }
      }

    } catch (IOException ex) {
      ex.printStackTrace();
    }
  }

  static public byte[] swab(byte[] buffer, int bytesRead) {
    byte[] buffer_swabbed = new byte[BUFFER_SIZE];
    // Swab bytes
    if(bytesRead % 2 == 0) {
      for(int i=0;i<bytesRead;i=i+2) {
        buffer_swabbed[i]=buffer[i+1];
        buffer_swabbed[i+1]=buffer[i];
      }
    } else {
      for(int i=0;i<bytesRead-1;i=i+2) {
        buffer_swabbed[i]=buffer[i+1];
        buffer_swabbed[i+1]=buffer[i];
      }
      buffer_swabbed[bytesRead-1] = buffer[bytesRead-1];
    }
    return buffer_swabbed;
  }
}
